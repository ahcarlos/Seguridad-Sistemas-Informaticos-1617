def fast_expo(b, m)
  # b -> exponente
  # m -> modulo
  x = 1
  y = self % m
  while b > 0 && y > 1
    if b.odd?
      x = (x * y) % m
      b -= 1
    else
      y *= y % m
      b /= 2
    end
  end
  x
end
